  <!-- JavaScript to Load Cards -->
  <script>
    const container = document.getElementById('card-collection');
    fetch('https://api.pokemontcg.io/v2/cards?pageSize=12')
      .then(res => res.json())
      .then(data => {
        container.innerHTML = ''; // Clear loading text
        data.data.forEach(card => {
          const img = document.createElement('img');
          img.src = card.images.large || card.images.small;
          img.alt = card.name;
          container.appendChild(img);
        });
      })
      .catch(err => {
        console.error("Failed to load cards:", err);
        container.innerHTML = '<p style="color:red;">Failed to load cards.</p>';
      });
  </script>
